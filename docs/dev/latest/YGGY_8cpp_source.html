104 void YGGY::define_tx() {
105  int vindex { 0 };
106  for (auto var: *vars) {
107  double lb { var->get(GRB_DoubleAttr_LB) };
108  double ub { var->get(GRB_DoubleAttr_UB) };
109  if (var->get(GRB_CharAttr_VType) == 'C') {
110  tx.push_back( new GRBVar( transformation.addVar(-GRB_INFINITY, GRB_INFINITY, 0.0, GRB_CONTINUOUS) ) );
111  transformation.update();
112  if (lb != -GRB_INFINITY)
113  transformation.addConstr(*(tx.back()) - lb*u >= 0.0); // get umin & use ?!
114  if (ub != GRB_INFINITY)
115  transformation.addConstr(*(tx.back()) - ub*u <= 0.0); // use umax ?!
116  // set start solution ... z_start = u_start * var_start
117  }
118  else {
119  tx.push_back( new GRBVar( transformation.addVar(lb, ub, 0.0, GRB_BINARY) ) );
120  if (start_solution) {
121  tx.back()->set(GRB_DoubleAttr_Start, (*start_solution)[vindex]);
122  }
123  }
124  vindex++;
125  }
126 }

140 void YGGY::define_objective(GRBLinExpr& objNumerator, int objSense) {
141  double d { objNumerator.getConstant() };
142  GRBQuadExpr transformed_objective { d*u };
143  for (unsigned int i = 0; i < objNumerator.size(); i++) {
144  GRBVar var = objNumerator.getVar(i);
145  int j = getIndex(var);
146  double coeff = objNumerator.getCoeff(i);
147  if (var.get(GRB_CharAttr_VType) == 'C')
148  transformed_objective += coeff * ( *tx[j] );
149  else
150  transformed_objective += coeff * u * ( *tx[j] );
151  }
152  transformation.setObjective(transformed_objective, objSense);
153 }

155 void YGGY::define_constraints(GRBLinExpr& objDenominator) {
156  double f { objDenominator.getConstant() };
157  GRBQuadExpr lhs { f * u };
158  for (unsigned int i = 0; i < objDenominator.size(); i++) {
159  GRBVar var = objDenominator.getVar(i);
160  int j = getIndex(var);
161  double coeff = objDenominator.getCoeff(i);
162  if (var.get(GRB_CharAttr_VType) == 'C')
163  lhs += coeff * ( *tx[j] );
164  else
165  lhs += coeff * u * ( *tx[j] );
166  }
167  transformation.addQConstr(lhs, GRB_EQUAL, 1.0);
168 
169 
170  GRBConstr* constrs { base_model->getConstrs() };
171  int numConstrs { base_model->get(GRB_IntAttr_NumConstrs) };
172  for (int k = 0; k < numConstrs; k++) {
173  GRBLinExpr origLhs { base_model->getRow(*constrs) };
174  double b { constrs->get(GRB_DoubleAttr_RHS) };
175  lhs = - b * u ;
176  for (unsigned int i = 0; i < origLhs.size(); i++) {
177  GRBVar var = origLhs.getVar(i);
178  int j = getIndex(var);
179  double coeff = origLhs.getCoeff(i);
180  if (var.get(GRB_CharAttr_VType) == 'C')
181  lhs += coeff * ( *tx[j] );
182  else
183  lhs += coeff * u * ( *tx[j] );
184  }
185  char constrSense { constrs->get(GRB_CharAttr_Sense) };
186  transformation.addQConstr(lhs, constrSense, 0.0);
187  constrs++;
188  }
189  transformation.update();
190 }

234 void YGGY::backTransformSolution() {
235  int status { transformation.get(GRB_IntAttr_Status) };
236  if (status == GRB_OPTIMAL || status == GRB_INTERRUPTED || status == GRB_TIME_LIMIT) {
237  solution.objVal = transformation.get(GRB_DoubleAttr_ObjVal);
238  double uval { u.get(GRB_DoubleAttr_X) };
239  for (int i = 0; i < base_model->get(GRB_IntAttr_NumVars); i++) {
240  double value { tx[i]->get(GRB_DoubleAttr_X) };
241  if (tx[i]->get(GRB_CharAttr_VType) == 'C')
242  solution.varVals.push_back( value / uval );
243  else
244  solution.varVals.push_back( value );
245  }
246  }
247  else std::cout << "GRB_Status : " << transformation.get(GRB_IntAttr_Status) << std::endl;
248 }
