111  int transform(GrbfrcCallback<T>* cb) {
112 
113  copy_model();
114 
115  u = transformation.addVar(*umin, *umax, 0.0, GRB_CONTINUOUS);
116  transformation.setObjective(GRBLinExpr(u), base_model->get(GRB_IntAttr_ModelSense));
117  transformation.update();
118  add_u_constraint();
119 
120  if (cb) {
121  callback = cb->yield(tx, vars);
122  transformation.setCallback(callback);
123  }
124 
125  transformation.update();
126  return 0;
