54  // check whether referenced variables are discrete
55  if (vargrps) {
56  transformed_vargrps = new std::vector<std::map<T, GRBVar>>();
57  for (std::map<T, GRBVar>& vargrp : *(vargrps)) {
58  std::map<T, GRBVar> transformed_vargrp;

63  this->register_vargrps();
