48 /*
49  * max (cc*xc + cd*xd + d)/(ec*xc + ed*xd + f)
50  *
51  * s.t.
52  *
53  * Ac*xc + Ad*xd = b
54  * xc in R
55  * xd in Z
56  */
57 
58 /*
59  * u = 1 / (ec*xc + ed*xd + f)
60  * z = u*xc = xc / (ec*xc + ed*xd + f)
61  *
62  */
63 
64 /*
65  * max (cc*z + cd*(u*xd) + d*u)
66  *
67  * s.t.
68  *
69  * Ac*z + Ad*(u*xd) - b*u = 0
70  * ec*z + ed*(u*xd) + f*u = 1
71  * z, u in R
72  * xd in Z
73  *
74  */
75

79  * max (cc*z + cd*w + d*u)
80  *
81  * s.t.
82  *
83  * Ac*z + Ad*w - b*u = 0
84  * ec*z + ed*w + f*u = 1
85  * GloverConstraints(w, xd, u)
86  * u,w,z in R
87  * xd in Z
88  *
89  */

113  template <typename T>
114  int transform(GrbfrcCallback<T>* cb) {
115  // denominator variable u = 1 / (dx + ey + f)
116  double umax = 0.0;
117  if (!getUmax(umax)) return 127;
118  u = transformation.addVar(0.0, umax, 0.0, GRB_CONTINUOUS);
119  // tx ~ (u*xc, xd)
120  std::cout << "Define tx ..." << std::endl;
121  define_tx();
122  std::cout << "Setting Callback ..." << std::endl;
123  if (cb) {
124  callback = cb->yield(tx, vars);
125  transformation.setCallback(callback);
126  }
127 
128  // handle more generically
129  transformation.set(GRB_IntParam_LazyConstraints, base_model->get(GRB_IntParam_LazyConstraints));
130 
131  transformation.update();
132  do_the_rest(umax);
133  return 0;
134  }
